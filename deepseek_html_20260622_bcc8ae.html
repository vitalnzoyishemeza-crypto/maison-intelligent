⚠ Diviseur de Tension HC-06

IMPORTANT : Le HC-06 fonctionne en 3.3V !

Arduino D3 (TX 5V) ───[ 1kΩ ]───┬─── HC-06 RX (3.3V) │ [ 2.2kΩ ] │ GND

🔌 Relais - Pompe/Lampe 220V

⚠ DANGER 220V - À faire vérifier par un électricien

Alim 220V Phase ─── COM Relais 1 ─── NO ─── Pompe (+) Alim 220V Neutre ────────────────────────── Pompe (-) Alim 220V Phase ─── COM Relais 2 ─── NO ─── Lampe (+) Alim 220V Neutre ────────────────────────── Lampe (-)

// ============================================= // 🏠 MAISON INTELLIGENTE PRO MAX - VERSION FINALE // Arduino UNO + HC-06 + MQ-135 + PIR + LDR // 2 Relais (Pompe Eau + Lampes) // ============================================= #include <SoftwareSerial.h> #include <EEPROM.h> // ===== PINS ===== const int pirPin = 4; const int ldrPin = A0; const int mq135Pin = A1; const int relaisPompe = 7; const int relaisLampe = 8; const int buzzerPin = 9; const int ledRouge = 12; const int ledVerte = 13; SoftwareSerial bluetooth(2, 3); // ===== STRUCTURE PARAMETRES (EEPROM) ===== struct Parametres { int seuilLDR = 500; int seuilGaz = 400; int dureePIR = 30; int pinCode = 1234; bool securiteActive = true; bool buzzerActive = true; byte heureOn = 18; byte heureOff = 6; byte minuteOn = 0; byte minuteOff = 0; }; Parametres params; // ===== VARIABLES ===== String commandeBT = ""; unsigned long dernierEnvoi = 0, dernierMouvement = 0, derniereLecture = 0; bool etatPompe = false, etatLampe = false, alarmeGaz = false, systemeVerrouille = true; enum Mode { MANUEL, LDR, PIR, AUTO, PROGRAMME }; Mode modeActuel = MANUEL; String modeNom[] = {"MANUEL", "LDR", "PIR", "AUTO", "PROGRAMME"}; int compteurPompe = 0, compteurLampe = 0; unsigned long dureePompe = 0, dureeLampe = 0, debutPompe = 0, debutLampe = 0; unsigned long dernierTickHorloge = 0; int heureActuelle = 12, minuteActuelle = 0; // ===== SETUP ===== void setup() { pinMode(pirPin, INPUT); pinMode(ldrPin, INPUT); pinMode(mq135Pin, INPUT); pinMode(relaisPompe, OUTPUT); pinMode(relaisLampe, OUTPUT); pinMode(buzzerPin, OUTPUT); pinMode(ledRouge, OUTPUT); pinMode(ledVerte, OUTPUT); digitalWrite(relaisPompe, LOW); digitalWrite(relaisLampe, LOW); digitalWrite(buzzerPin, LOW); Serial.begin(9600); bluetooth.begin(9600); chargerParametres(); animationDemarrage(); bluetooth.println("SYSTEME PRO MAX PRET"); bluetooth.print("VERROUILLE:OUI|PIN:"); bluetooth.println(params.pinCode); digitalWrite(ledVerte, HIGH); } // ===== LOOP ===== void loop() { miseAJourHorloge(); if (millis() - derniereLecture > 200) { int ldr = analogRead(ldrPin), gaz = analogRead(mq135Pin), pir = digitalRead(pirPin); verifierSecuriteGaz(gaz); if (!alarmeGaz && !systemeVerrouille) { switch(modeActuel) { case LDR: modeLDR(ldr); break; case PIR: modePIR(pir); break; case AUTO: modeAuto(ldr, pir); break; case PROGRAMME: modeProgramme(); break; } } derniereLecture = millis(); } if (millis() - dernierEnvoi > 1000 && !systemeVerrouille) { envoyerEtatComplet(); dernierEnvoi = millis(); } lireBluetooth(); if (alarmeGaz) { digitalWrite(ledRouge, millis() % 500 < 250); digitalWrite(ledVerte, LOW); } else if (systemeVerrouille) { digitalWrite(ledRouge, millis() % 2000 < 100); digitalWrite(ledVerte, LOW); } else { digitalWrite(ledRouge, LOW); digitalWrite(ledVerte, HIGH); } delay(10); } // ... (modes LDR, PIR, AUTO, PROGRAMME, sécurité gaz, Bluetooth, EEPROM) // Le code complet est inclus dans le téléchargement .ino
